def to_hash
  if @style == :link
    {
      type: 2,
      label: @label,
      style: self.class.styles[@style],
      url: @url,
      emoji: hash_emoji(@emoji),
      disabled: @disabled,
    }
  else
    {
      type: 2,
      label: @label,
      style: self.class.styles[@style],
      custom_id: @custom_id,
      emoji: hash_emoji(@emoji),
      disabled: @disabled,
    }
  end
end
