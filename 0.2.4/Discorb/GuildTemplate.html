def edit(name = nil, description = :unset)
  Async do
    payload = {}
    payload[:name] = name if name
    payload[:description] = description if description != :unset
    @client.http.patch("/guilds/#{@source_guild_id}/templates/#{@code}", payload).wait
  end
end
